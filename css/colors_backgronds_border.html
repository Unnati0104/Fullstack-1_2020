Built-In Color

       h2 {
           color: red;
       }

RGB Format

       h2 {
           color: rgb(0,255,150);
       }

RGBA Format

       h2 {
           color: rgba(0,255,150,0.5);
       }

Hexadecimal Notation

       h2 {
           color: #000FFF;
       }

HSL(hue, saturation, lightness)

       h2 {
           color: hsl(360,50%,100%);
       }

HSLA(hue, saturation, lightness, alpha)

       h2 {
           color: hsla(360,50%,100%,0.5);
       }

Text Color

       h2 {
           color: color-name;
       }

Background Color

       h2 {
           color: color-name;
       }

Border Color

       h2 {
           color: color-name;
       }

Border Radius

       h2 {
           border-radius: 60%
       }

Background Color and Opacity

       h2 {
           background-color: color-name;
           opacity: 0.5;
       }

Background Image

       h2 {
           background-image: url('address of image');
       }

Background Repeat

       h2 {
           background-repeat: no-repeat;
       }

Background Position

       h2 {
           background-position: value;
       }

Background Size

       h2 {
           background-size:  auto|length|cover|contain|initial|inherit; 
       }

Background Attachment

       h2 {
           background-attachment:  scroll|fixed|local|initial|inherit; 
       }
