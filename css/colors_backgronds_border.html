Built-In Color

       h2 {
           color: red;
       }

RGB Format

       h2 {
           color: rgb(0,255,150);
       }

RGBA Format

       h2 {
           color: rgba(0,255,150,0.5);
       }

Hexadecimal Notation

       h2 {
           color: #000FFF;
       }

HSL(hue, saturation, lightness)

       h2 {
           color: hsl(360,50%,100%);
       }

HSLA(hue, saturation, lightness, alpha)

       h2 {
           color: hsla(360,50%,100%,0.5);
       }
